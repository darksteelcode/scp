//Comment-This is a basic format fro the microcde file
0:start:MAR_MUX_SEL=1,DECODE_NEXT_SEL=2,DECODE_NEXT=256//This loads the memory, next decodes it to start off the processor
1:ldia:PC_INC, MAR_MUX_SEL=1, DECODE_NEXT_SEL=2,DECODE_NEXT=257 //Increment PC and set MAR equal to it

256:start2:DECODE_NEXT_SEL=0 //Decode Next

257:ldia_2:MDR_OUT_BYTE_LOW_WE, DECODE_NEXT_SEL=2,DECODE_NEXT=258 //Load mem into MDR
258:ldia_3:AB_BUS_IN_SEL=1,A_BUS_WE//move MDR to A reg, and fetch next, and set to decode
